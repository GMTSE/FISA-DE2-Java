  \begin{itemize}
    \item<1-> Why? Eclipse fine, but does not transfer local config!

% 1. You compile in Eclipse locally => fine (Eclipse internal compiler)
% 2. You send your code to a colleague => KO
% 3. PB: Eclipse's compiler options, Classpath, Buildpath, etc are not transfered :{ & you don't remember what you've changed!
    \item<2-> Goal: Manage compilation chain \& dependencies
    \bigskip
    \item<3-> Uses conventions\\
    \onslide<4-5>{
      \medskip
        \begin{itemize}
          \item<4-5> Project architecture
          \medskip
          \item<5> Pre-configured rules:\\
          \texttt{clean}, \texttt{compile}\ldots{}
        \end{itemize}
    }
    \bigskip
    \item<6-> Configuration via \texttt{pom.xml} file
    \bigskip
    \item<7-> Example usage:\\
    \texttt{mvn clean compile}
    \bigskip
    \item<7-> We will let Eclipse do (most of) that for us!!
  \end{itemize}

}
\definecolor{gray}{rgb}{0.4,0.4,0.4}
\definecolor{darkblue}{rgb}{0.0,0.0,0.6}
\definecolor{cyan}{rgb}{0.0,0.6,0.6}

  \begin{lstlisting}[language=XML]
  <project>
    <!-- model version is always 4.0.0 for Maven 2.x POMs -->
    <modelVersion>4.0.0</modelVersion>
    <!-- project coordinates, i.e. a group of values which uniquely identify this project -->
    <groupId>com.mycompany.app</groupId>
    <artifactId>my-app</artifactId>
    <version>1.0</version>
    <!-- library dependencies -->
    <dependencies>
      <dependency>
        <!-- coordinates of the required library -->
        <groupId>junit</groupId>
        <artifactId>junit</artifactId>
        <version>3.8.1</version>
        <!-- this dependency is only used for running and compiling tests -->
        <scope>test</scope>
      </dependency>
    </dependencies>
  </project>
  \end{lstlisting}

    \medskip
    \item \lstinline[language=XML]{groupId} $\approx$ package
    \item \lstinline[language=XML]{artifactId} $\approx$ main class
\end{itemize}
